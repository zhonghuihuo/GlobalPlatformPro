022package openkms.gp;
023
024import java.io.File;
025import java.security.NoSuchAlgorithmException;
026import java.util.List;
027
028import javax.smartcardio.Card;
029import javax.smartcardio.CardException;
030import javax.smartcardio.CardTerminal;
031import javax.smartcardio.CardTerminals;
032import javax.smartcardio.CardTerminals.State;
033import javax.smartcardio.TerminalFactory;
034
035/**
036 * Facilitates working with javax.smartcardio
037 *
038 * @author Martin Paljak
039 *
040 */
041public class TerminalManager {
042
043        private static final String debian_path = "/usr/lib/libpcsclite.so.1";
044        private static final String ubuntu_path = "/lib/libpcsclite.so.1";
045
046        private static boolean buggy = true;
047
048        public static TerminalFactory getTerminalFactory() throws NoSuchAlgorithmException {
049                return getTerminalFactory(true);
050        }
051
052        public static TerminalFactory getTerminalFactory(boolean fix) throws NoSuchAlgorithmException {
053                // Set necessary parameters for seamless PC/SC access. OpenJDK has wrong
054                // paths (without .1) See this blog post:
055                // http://ludovicrousseau.blogspot.com.es/2013/03/oracle-javaxsmartcardio-failures.html
056                if (System.getProperty("os.name").equalsIgnoreCase("Linux")) {
057                        if (new File(debian_path).exists()) {
058                                System.setProperty("sun.security.smartcardio.library", debian_path);
059                        } else if (new File(ubuntu_path).exists()) {
060                                System.setProperty("sun.security.smartcardio.library", ubuntu_path);
061                        }
062                }
063
064                TerminalFactory tf = TerminalFactory.getDefault();
065                // OSX is horribly broken. Use JNA based approach if not already
066                // installed and used as default
067                if (fix) {
068                        if (System.getProperty("os.name").equalsIgnoreCase("Mac OS X")) {
069                                if (tf.getProvider().getName() != jnasmartcardio.Smartcardio.PROVIDER_NAME) {
070                                        tf = TerminalFactory.getInstance("PC/SC", null, new jnasmartcardio.Smartcardio());
071                                }
072                        }
073                }
074
075                // Right now only JNA based approach should be correct.
076                if (tf.getProvider().getName() == jnasmartcardio.Smartcardio.PROVIDER_NAME) {
077                        buggy = false;
078                }
079                return tf;
080        }
081
082        /**
083         * Calls {@link javax.smartcardio.Card#disconnect(boolean)} with the fixed reset parameter.
084         *
085         * The parameter is fixed based on the used provider and assumes that written code is correct.
086         *
087         * @param card The card on what to act
088         * @param reset The intended operation after disconnect
089         * @throws CardException
090         */
091        public static void disconnect(Card card, boolean reset) throws CardException {
092                card.disconnect(buggy ? !reset : reset);
093        }
094
095        public static CardTerminal getTheReader() throws CardException {
096                try {
097                        TerminalFactory tf = getTerminalFactory();
098                        CardTerminals tl = tf.terminals();
099                        List<CardTerminal> list = tl.list(State.CARD_PRESENT);
100                        if (list.size() == 0) {
101                                // No readers with cards. Maybe empty readers?
102                                list = tl.list(State.ALL);
103                        }
104
105                        if (list.size() != 1) {
106                                throw new RuntimeException("This application expects one and only one card reader (with an inserted card)");
107                        } else {
108                                return tl.list().get(0);
109                        }
110                } catch (NoSuchAlgorithmException e) {
111                        throw new CardException(e);
112                }
113        }
114
115        // Given an instance of some Exception from a PC/SC system,
116        // return a meaningful PC/SC error name.
117        public static String getExceptionMessage(Exception e) {
118                String classname = e.getClass().getCanonicalName();
119                if (e instanceof CardException) {
120                        // This comes from SunPCSC most probably and already contains the PC/SC error in the cause
121                        if (e.getCause() != null) {
122                                if (e.getCause().getMessage() != null) {
123                                        if (e.getCause().getMessage().indexOf("SCARD_") != -1) {
124                                                return e.getCause().getMessage();
125                                        }
126                                }
127                        }
128                }
129                if (classname != null && classname.equalsIgnoreCase("jnasmartcardio.Smartcardio.EstablishContextException")) {
130                        if (e.getCause().getMessage().indexOf("SCARD_E_NO_SERVICE") != -1)
131                                return "SCARD_E_NO_SERVICE";
132                }
133                return null;
134        }
135}
