023package openkms.gpj;
024
025/**
026 * Root exception class for all global platform protocol errors.
027 */
028public class GPException extends Exception {
029
030        private static final long serialVersionUID = -642613357615559636L;
031
032        /**
033         * Response status indicating the error, or 0 if not applicable.
034         */
035        public final int sw;
036
037        /**
038         *
039         * Constructs a new GPException with the specified detail message.
040         *
041         * @param sw
042         *            failing response status
043         * @param message
044         *            the detailed message
045         */
046        public GPException(int sw, String message) {
047                super(message + " SW: " + GPUtils.swToString(sw));
048                this.sw = sw;
049        }
050
051        public GPException(String message) {
052                super(message);
053                this.sw = 0x0000;
054        }
055}
